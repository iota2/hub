170  /* Initialize SPI interface */
171  i2_spi_init(&ssd1306);
172 
173  /* Start Initialization by bringing RESET HIGH */
174  i2_gpio_set(&ssd1306_RST, I2_HIGH);
175  /* VDD (3.3V) goes high at start, lets just chill for a ms */
176  ssd1306_delay(2);
177 
178  /* Bring RESET LOW */
179  i2_gpio_set(&ssd1306_RST, I2_LOW);
180  ssd1306_delay(1);

184  ssd1306_delay(2);

277 /**
278  * @brief Draws a single pixel in image buffer.
279  * @details Updates corresponding Layer buffer with pixel value.
280  *
281  * @param[in] y Vertical (y) position (0..63).
282  * @param[in] x Horizontal (x) position (0..127).
283  * @param[in] color Color of line to draw @ref SSD1306_COLOR_SPEC.
284  * @param[in] layer Layer to draw line @ref SSD1306_MULTI_LAYER_SPEC.
285  * @return None.
286  */
287 void ssd1306_draw_pixel(int16_t y, int16_t x, uint16_t color, uint16_t layer)
288 {
289  if ((x >= SSD1306_DISPLAY_WIDTH) ||
290  (x < 0) ||
291  (y >= SSD1306_DISPLAY_HEIGHT) ||
292  (y < 0)) {
293  return;
294  }
295 
296  if (layer & SSD1306_LAYER1) {
297  switch (color) {
298  case SSD1306_WHITE :
299  ssd1306_buffer_layer1[x+ (y/8)*SSD1306_DISPLAY_WIDTH] |= (1 << (y&7));
300  break;
301  case SSD1306_BLACK :
302  ssd1306_buffer_layer1[x+ (y/8)*SSD1306_DISPLAY_WIDTH] &= ~(1 << (y&7));
303  break;
304  case SSD1306_INVERSE :
305  ssd1306_buffer_layer1[x+ (y/8)*SSD1306_DISPLAY_WIDTH] ^= (1 << (y&7));
306  break;
307  }
308  }
309 
310 #if ( SSD1306_MULTILAYER_SUPPORT == I2_ENABLE )
311  if (layer & SSD1306_LAYER2) {
312  switch (color) {
313  case SSD1306_WHITE :
314  ssd1306_buffer_layer2[x+ (y/8)*SSD1306_DISPLAY_WIDTH] |= (1 << (y&7));
315  break;
316  case SSD1306_BLACK :
317  ssd1306_buffer_layer2[x+ (y/8)*SSD1306_DISPLAY_WIDTH] &= ~(1 << (y&7));
318  break;
319  case SSD1306_INVERSE :
320  ssd1306_buffer_layer2[x+ (y/8)*SSD1306_DISPLAY_WIDTH] ^= (1 << (y&7));
321  break;
322  }
323  }
324 #endif /* SSD1306_MULTILAYER_SUPPORT */
325 
326 #if 0
327  ssd1306_refresh();
328 #endif /* 0 */
329 }

452 /**
453  * @brief Draws a straight line from point A to point B.
454  * @details Draws a line between two coordinates.
455  *
456  * @param[in] y0 Vertical (y) position for beginning point A.
457  * @param[in] x0 Horizontal (x) position for beginning point A.
458  * @param[in] y1 Vertical (y) position for terminating point B.
459  * @param[in] x1 Horizontal (x) position for terminating point B.
460  * @param[in] color Color of line to draw @ref SSD1306_COLOR_SPEC.
461  * @param[in] layer Layer to draw line @ref SSD1306_MULTI_LAYER_SPEC.
462  * @return None.
463  */
464 void ssd1306_draw_line(int16_t y0, int16_t x0, int16_t y1, int16_t x1,
465  uint16_t color, uint16_t layer)
466 {
467  int16_t steep = (abs(y1 - y0) > abs(x1 - x0));
468  int16_t dx = 0, dy = 0;
469  int16_t err = 0;
470  int16_t ystep = 0;
471 
472  if (steep) {
473  SSD1306_SWAP(x0, y0);
474  SSD1306_SWAP(x1, y1);
475  }
476 
477  if (x0 > x1) {
478  SSD1306_SWAP(x0, x1);
479  SSD1306_SWAP(y0, y1);
480  }
481 
482  dx = x1 - x0;
483  dy = abs(y1 - y0);
484  err = dx / 2;
485 
486  if (y0 < y1) {
487  ystep = 1;
488  } else {
489  ystep = -1;
490  }
491 
492  for (; x0<=x1; x0++) {
493  if (steep) {
494  ssd1306_draw_pixel(x0, y0, color, layer);
495  } else {
496  ssd1306_draw_pixel(y0, x0, color, layer);
497  }
498  err -= dy;
499  if (err < 0) {
500  y0 += ystep;
501  err += dx;
502  }
503  }
504 }

576  * @brief Draws a filled polygon.
577  * @details http://alienryderflex.com/polygon_fill/.
578  *
579  * @param[in] poly SSD1306 polygon parameter @ref SSD1306_POLYGON_SPEC.
580  * @param[in] y (y) coordinate to draw.
581  * @param[in] x (x) coordinate to draw.
582  * @param[in] color Color of polygon to draw @ref SSD1306_COLOR_SPEC.
583  * @param[in] layer Layer to draw polygon @ref SSD1306_MULTI_LAYER_SPEC.
584  * @return None.
585  */
586 void ssd1306_fill_polygon(ssd1306_poly_t* poly, double y, double x,
587  uint16_t color, uint16_t layer)
588 {
589  int nodeX[SSD1306_MAX_POLY_CORNERS], nodes, i, j, swap;
590  double polyX[SSD1306_MAX_POLY_CORNERS], pixelX;
591  double polyY[SSD1306_MAX_POLY_CORNERS], pixelY;
592 
593  /* draw shape */
594  ssd1306_draw_polygon(poly, (int16_t)y, (int16_t)x, color, layer);
595  /* copy polygon */
596  i = poly->corners - 1;
597  while (i >= 0) {
598  polyX[i] = round(poly->point[i].x + x);
599  polyY[i] = round(poly->point[i].y + y);
600  i--;
601  }
602  /* Loop through the rows of the image */
603  for (pixelY=0; pixelY<SSD1306_DISPLAY_HEIGHT; pixelY++) {
604  /* Build a list of nodes */
605  nodes = 0;
606  j = poly->corners - 1;
607  for (i=0; i<poly->corners; i++) {
608  if ((polyY[i]<(double) pixelY && polyY[j]>=(double) pixelY) ||
609  (polyY[j]<(double) pixelY && polyY[i]>=(double) pixelY)) {
610  nodeX[nodes++] = (int)round( polyX[i] +
611  (pixelY-polyY[i])/(polyY[j]-polyY[i]) *
612  (polyX[j]-polyX[i]) );
613  }
614  j = i;
615  }
616 
617  /* Sort the nodes, via a simple Bubble sort */
618  i = 0;
619  while (i<nodes-1) {
620  if (nodeX[i]>nodeX[i+1]) {
621  swap = nodeX[i];
622  nodeX[i] = nodeX[i+1];
623  nodeX[i+1] = swap;
624  if (i) {
625  i--;
626  }
627  } else {
628  i++;
629  }
630  }
631 
632  /* Fill the pixels between node pairs. */
633  for (i=0; i<nodes; i+=2) {
634  if (nodeX[i]>=SSD1306_DISPLAY_WIDTH) {
635  break;
636  }
637  if (nodeX[i+1]> 0) {
638  if (nodeX[i]< 0) {
639  nodeX[i] = 0 ;
640  }
641  if (nodeX[i+1] > SSD1306_DISPLAY_WIDTH) {
642  nodeX[i+1] = SSD1306_DISPLAY_WIDTH;
643  }
644  for (pixelX=nodeX[i]; pixelX<=nodeX[i+1]; pixelX++) {
645  ssd1306_draw_pixel((int16_t)(pixelY), (int16_t)(pixelX), color, layer);
646  }
647  }
648  }
649  }
650 }

667  * @brief Draws a circle.
668  * @details Draws a vertical line from a beginning coordinate to certain length.
669  *
670  * @param[in] y0 (y) coordinate of circle's center.
671  * @param[in] x0 (x) coordinate of circle's center.
672  * @param[in] r radius of circle to draw.
673  * @param[in] color Color of circle to draw @ref SSD1306_COLOR_SPEC.
674  * @param[in] layer Layer to draw circle @ref SSD1306_MULTI_LAYER_SPEC.
675  * @return None.
676  */
677 void ssd1306_draw_circle( uint8_t y0, uint8_t x0, uint8_t r,
678  uint16_t color, uint16_t layer)
679 {
680  int x = r;
681  int y = 0;
682  /* Decision criterion divided by 2 evaluated at x=r, y=0 */
683  int check_point = 1 - x;
684 
685  if (x < 0) {
686  return;
687  }
688 
689  while (y <= x) {
690  ssd1306_draw_pixel( y + y0, x + x0, color, layer); /* Octant 1 */
691  ssd1306_draw_pixel( x + y0, y + x0, color, layer); /* Octant 2 */
692  ssd1306_draw_pixel( y + y0, -x + x0, color, layer); /* Octant 4 */
693  ssd1306_draw_pixel( x + y0, -y + x0, color, layer); /* Octant 3 */
694  ssd1306_draw_pixel(-y + y0, -x + x0, color, layer); /* Octant 5 */
695  ssd1306_draw_pixel(-x + y0, -y + x0, color, layer); /* Octant 6 */
696  ssd1306_draw_pixel(-y + y0, x + x0, color, layer); /* Octant 8 */
697  ssd1306_draw_pixel(-x + y0, y + x0, color, layer); /* Octant 7 */
698  y++;
699 
700  if (check_point <= 0) {
701  /* Change in decision criterion for y -> y+1 */
702  check_point += 2 * y + 1;
703  } else {
704  x--;
705  /* Change for y -> y+1, x -> x-1 */
706  check_point += 2 * (y - x) + 1;
707  }
708  }
709 }
