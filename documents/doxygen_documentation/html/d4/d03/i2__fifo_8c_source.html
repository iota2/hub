85 /**
86  * @brief Get count of FIFO buffer.
87  * @details Returns the difference between write and read buffer.
88  *
89  * @param[in] fifo FIFO context @ref i2_fifo_t.
90  * @param[in] in_isr Is this function is called from an ISR or not.
91  * @return The difference between size of write and read buffers.
92  */
93 int32_t i2_fifo_count(i2_fifo_t *fifo, bool in_isr)
94 {
95  int32_t count;
96 
97 #if defined ( ENABLE_RTOS_AWARE_HAL )
98  if (!in_isr) {
99  vPortEnterCritical();
100  }
101 #else
102  UNUSED_PARAMETER(in_isr);
103 #endif
104 
105  count = fifo->wr_index - fifo->rd_index;
106  if (count < 0) {
107  count += fifo->fifo_size;
108  }
109 
110  if (count < fifo->fifo_size) {
111  fifo->wr_index %= fifo->fifo_size;
112  fifo->rd_index %= fifo->fifo_size;
113  }
114 
115 #if defined ( ENABLE_RTOS_AWARE_HAL )
116  if (!in_isr) {
117  vPortExitCritical();
118  }
119 #endif
120 
121  return count;
122 }

124 /**
125  * @brief Write data to FIFO buffer.
126  * @details Write data to FIFO.
127  *
128  * @param[in] fifo FIFO context to write @ref i2_fifo_t.
129  * @param[in] data Data to write on FIFO.
130  * @param[in] in_isr Is this function is called from an ISR or not.
131  * @return If Success returns FIFO count else ( -1 ) when error occurred.
132  */
133 int32_t i2_fifo_write(i2_fifo_t *fifo, uint8_t data, bool in_isr)
134 {
135  int32_t retval = 0;
136 
137  if (i2_fifo_count(fifo, in_isr) < fifo->fifo_size) {
138  fifo->buf[fifo->wr_index % fifo->fifo_size] = data;
139  fifo->wr_index++;
140  retval = i2_fifo_count(fifo, in_isr);
141  } else {
142  /* FIFO Full */
143  retval = -1;
144  }
145 
146  return retval;
147 }

149 /**
150  * @brief Read data from FIFO buffer.
151  * @details Read data from FIFO.
152  *
153  * @param[in] fifo FIFO context to read @ref i2_fifo_t.
154  * @param[in] *data Data read from FIFO.
155  * @param[in] in_isr Is this function is called from an ISR or not.
156  * @return If Success returns FIFO count else ( -1 ) when error occurred.
157  */
158 int32_t i2_fifo_read(i2_fifo_t *fifo, uint8_t *data, bool in_isr)
159 {
160  int32_t retval = 0;
161 
162  if ( i2_fifo_count(fifo, in_isr) ) {
163  *data = fifo->buf[fifo->rd_index % fifo->fifo_size];
164  fifo->rd_index++;
165  retval = i2_fifo_count(fifo, in_isr);
166  } else {
167  /* FIFO empty */
168  retval = -1;
169  }
170 
171  return retval;
172 }

174 /**
175  * @brief Read a buffer from FIFO.
176  * @details Reads specified bytes count from FIFO and & copy it to destination.
177  *
178  * @param[in] fifo FIFO context to read @ref i2_fifo_t.
179  * @param[in] dst Destination to copy buffer.
180  * @param[in,out] size Size to read form buffer.
181  * Updated size of bytes read from FIFO is returned.
182  * @param[in] in_isr Is this function is called from an ISR or not.
183  * @return None.
184  */
185 void i2_fifo_copy(i2_fifo_t *fifo, uint8_t *dst, int32_t *size, bool in_isr)
186 {
187  uint8_t data;
188  int32_t counter = 0;
189 
190  /* Read until given buffer is full or fifo empty */
191  while ((*size > counter) && (i2_fifo_read(fifo, &data, in_isr) >= 0)) {
192  *dst++ = data;
193  counter++;
194  }
195  *size = counter;
196 
197  return;
198 }
