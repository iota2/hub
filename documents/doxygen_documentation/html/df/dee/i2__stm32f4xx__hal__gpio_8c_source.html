342 /**
343  * @brief GPIO Pin initialization.
344  * @details Initializes a GPIO pin from instance.
345  *
346  * @param[in] *inst GPIO instance.
347  * @return None.
348  */
349 static void gpio_activate(i2_gpio_inst_t *inst)
350 {
351  GPIO_InitTypeDef config;
352 
353  if (inst->active) {
354  return;
355  }
356 
357  if ( gpio_refcount_get(inst->gpio_port) == 0 ) {
358  gpio_clock_enable(inst->gpio_port);
359  }
360 
361  config.Pin = inst->gpio;
362  config.Pull = inst->pull;
363  config.Mode = inst->mode;
364  config.Alternate = inst->alt;
365  config.Speed = inst->speed;
366  HAL_GPIO_Init(inst->gpio_port, &config);
367 
368  gpio_refcount_up(inst->gpio_port);
369  inst->active = true;
370 }

372 /**
373  * @brief GPIO Pin deinitialization.
374  * @details Deinitializes a GPIO pin from instance.
375  *
376  * @param[in] *inst GPIO instance.
377  * @return None.
378  */
379 static void gpio_deactivate(i2_gpio_inst_t *inst)
380 {
381  if (!inst->active) {
382  return;
383  }
384 
385  gpio_refcount_down(inst->gpio_port);
386  HAL_GPIO_DeInit(inst->gpio_port, inst->gpio);
387 
388  if (gpio_refcount_get(inst->gpio_port) == 0) {
389  gpio_clock_disable(inst->gpio_port);
390  }
391 
392  inst->active = false;
393 }

637 /**
638  * @brief GPIO interrupt configuration.
639  * @details Enables EXTI interrupt on a GPIO instance with specified callback.
640  *
641  * @param[in] *inst GPIO instance to use.
642  * @param[in] mode GPIO mode configuration.
643  * @param[in] pull GPIO pull / push configuration.
644  * @param[in] cb Callback for GPIO interrupt handler.
645  * @param[in] arg Arguments to be passed with callback.
646  * @retval None.
647  */
648 i2_error i2_gpio_config_interrupt(i2_gpio_inst_t *inst, uint32_t mode,
649  uint32_t pull, void (*cb)(void *arg), void *arg)
650 {
651  int32_t index;
652  IRQn_Type irqn;
653 
654  index = get_gpio_index(inst->gpio);
655 
656  /* Invalid gpio */
657  if ( index >= NUM_GPIO_PER_PORT ) {
658  return I2_INVALID_PARAM;
659  }
660 
661  /* Interrupt already registered for this line */
662  if ( gpio_isr_inst[index].cb ) {
663  return I2_NOT_AVAILABLE;
664  }
665 
666  if ( index == 0 ) {
667  irqn = EXTI0_IRQn;
668  } else if ( index == 1 ) {
669  irqn = EXTI1_IRQn;
670  } else if ( index == 2 ) {
671  irqn = EXTI2_IRQn;
672  } else if ( index == 3 ) {
673  irqn = EXTI3_IRQn;
674  } else if ( index == 4 ) {
675  irqn = EXTI4_IRQn;
676  } else if ( index >= 5 && index <= 9 ) {
677  irqn = EXTI9_5_IRQn;
678  } else if ( index >= 10 && index <= 15 ) {
679  irqn = EXTI15_10_IRQn;
680  } else {
681  return I2_INVALID_PARAM;
682  }
683  i2_gpio_config(inst, mode, pull);
684 
685  HAL_NVIC_SetPriority(irqn,
686  GPIO_EXTI_PREEMPTION_PRIORITY,
687  GPIO_EXTI_SUB_PRIORITY);
