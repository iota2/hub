42 /**
43  * @brief Function to call when a system assert call happens.
44  * @details Here write the operations when assert happens.
45  *
46  * @return None.
47  */
48 static void assert(void)
49 {
50  while (1) {
51  /*
52  * If error, toggle both LED's fast.
53  * The i value was experimentally determined
54  * based on the current 80Mhz core clock.
55  */
56  volatile int32_t i = 0;
57 
58  i2_led_on(LED1);
59  i2_led_off(LED2);
60  for (i = 0; i < 0x8ffff; i++) {
61  if (i == 0x8fffe) {
62  i2_led_toggle(LED1);
63  i2_led_toggle(LED2);
64  }
65  }
66  }
67 }
