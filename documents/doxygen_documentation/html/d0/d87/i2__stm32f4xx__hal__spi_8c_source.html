345  if (!strcmp(i2_spi_ctx_table[i].ctx_name, name)) {

350 }

371 /* Public functions ----------------------------------------------------------*/
372 /**
373  * @brief SPI initialization.
374  * @details Initializes a SPI instance along with GPIO, DMA and interrupts.
375  *
376  * @param[in] *inst SPI instance to initialize.
377  * @return Initialization error code @ref I2_ERROR.
378  */
379 i2_error i2_spi_init(i2_spi_inst_t *inst)
380 {
381  i2_error retval = I2_SUCCESS;
382  i2_spi_ctx_t *ctx;
383  SPI_HandleTypeDef *hspi;
384  SPI_TypeDef *base;
385  IRQn_Type irqn;
386  int32_t i;
387  uint32_t alt;
388 
389  if ( !inst ) {
390  return I2_INVALID_PARAM;
391  }
392 
393  /* get SPI context from name */
394  ctx = spi_inst_to_ctx(inst->ctx_name);
395  if ( !ctx ) {
396  return I2_INVALID_PARAM;
397  }
398 
399 #if defined ( ENABLE_RTOS_AWARE_HAL )
400  /* Create mutexes. */
401  ctx->mutex = xSemaphoreCreateMutex();
402 
403  /* Create semaphores */
404  ctx->sem = xSemaphoreCreateBinary();
405 #endif /* ENABLE_RTOS_AWARE_HAL */
406 
407  /* SPI driver needs GPIO */
408  i2_gpio_init();
409 
410  /* Configure CS */
411  if (i2_gpio_is_valid(&inst->CS)) {
412  i2_gpio_config_out(&inst->CS, I2_HIGH);
413  }
414 
415  /*
416  * If context has been already initialized,
417  * only init SPI device instance specific portions.
418  */
419  if ( ctx->initialized ) {
420  for (i = 0; i < I2_MAX_NUM_SPI_INSTANCE; i++) {
421  if (!i2_spi_inst_table[i]) {
422  i2_spi_inst_table[i] = inst;
423  break;
424  }
425  }
426  return I2_SUCCESS;
427  }
428 
429  /* Get the pointer to the SPI handle */
430  hspi = &(ctx->spi);
431 
432  /* Get the SPI controller base address */
433  base = ctx->baseaddr;
434 
435  ctx->status = I2_TRANSFER_DONE;
436 
437  if ( base == SPI1 ) {
438  spi1 = hspi;
439  irqn = SPI1_IRQn;
440  alt = GPIO_AF5_SPI1;
441  __HAL_RCC_SPI1_CLK_ENABLE();
442  if ( ctx->hal_mode == DMA_MODE ) {
443  __HAL_RCC_DMA2_CLK_ENABLE();
444  }
445  } else if ( base == SPI2 ) {
446  spi2 = hspi;
447  irqn = SPI2_IRQn;
448  alt = GPIO_AF5_SPI2;
449  __HAL_RCC_SPI2_CLK_ENABLE();
450  if ( ctx->hal_mode == DMA_MODE ) {
451  __HAL_RCC_DMA1_CLK_ENABLE();
452  }
453  } else if ( base == SPI3 ) {
454  spi3 = hspi;
455  irqn = SPI3_IRQn;
456  alt = GPIO_AF6_SPI3;
457  __HAL_RCC_SPI3_CLK_ENABLE();
458  if ( ctx->hal_mode == DMA_MODE ) {
459  __HAL_RCC_DMA1_CLK_ENABLE();
460  }
461  } else {
462  return I2_INVALID_PARAM;
463  }
464 
465  /* Configure SCLK */
466  if ( i2_gpio_is_valid(&ctx->SCLK) ) {
467  i2_gpio_config_alt(&ctx->SCLK, GPIO_MODE_AF_PP, GPIO_PULLDOWN, alt);
468  }
469 
470  /* Configure MISO */
471  if ( i2_gpio_is_valid(&ctx->MISO) ) {
472  i2_gpio_config_alt(&ctx->MISO, GPIO_MODE_AF_PP, GPIO_PULLDOWN, alt);
473  }
474 
475  /* Configure MOSI */
476  if ( i2_gpio_is_valid(&ctx->MOSI) ) {
477  i2_gpio_config_alt(&ctx->MOSI, GPIO_MODE_AF_PP, GPIO_PULLDOWN, alt);
478  }
479 
480  /* DMA RX Config */
481  if ( ctx->hal_mode == DMA_MODE) {
482  /*##### Configure the DMA streams ########################################*/
483  /* Configure the DMA handler for Transmission process */
484  ctx->hdma_rx.Instance = ctx->dma_rx_stream;
485  ctx->hdma_rx.Init.Channel = ctx->dma_rx_channel;
486  ctx->hdma_rx.Init.Direction = DMA_PERIPH_TO_MEMORY;
487  ctx->hdma_rx.Init.PeriphInc = DMA_PINC_DISABLE;
488  ctx->hdma_rx.Init.MemInc = DMA_MINC_ENABLE;
489  ctx->hdma_rx.Init.PeriphDataAlignment = DMA_PDATAALIGN_BYTE;
490  ctx->hdma_rx.Init.MemDataAlignment = DMA_MDATAALIGN_BYTE;
491  ctx->hdma_rx.Init.Mode = DMA_NORMAL;
492  ctx->hdma_rx.Init.Priority = ctx->dma_rx_priority;
493  ctx->hdma_rx.Init.FIFOMode = DMA_FIFOMODE_DISABLE;
494  ctx->hdma_rx.Init.FIFOThreshold = DMA_FIFO_THRESHOLD_FULL;
495  ctx->hdma_rx.Init.MemBurst = DMA_MBURST_SINGLE;
496  ctx->hdma_rx.Init.PeriphBurst = DMA_MBURST_SINGLE;
497 
498  HAL_DMA_Init(&ctx->hdma_rx);
499 
500  /* Associate the initialized DMA handle to the the SPI handle */
501  __HAL_LINKDMA(hspi, hdmarx, ctx->hdma_rx);
502 
503  /* Configure the DMA handler for Transmission process */
504  ctx->hdma_tx.Instance = ctx->dma_tx_stream;
505  ctx->hdma_tx.Init.Channel = ctx->dma_tx_channel;
506  ctx->hdma_tx.Init.Direction = DMA_MEMORY_TO_PERIPH;
507  ctx->hdma_tx.Init.PeriphInc = DMA_PINC_DISABLE;
508  ctx->hdma_tx.Init.MemInc = DMA_MINC_ENABLE;
509  ctx->hdma_tx.Init.PeriphDataAlignment = DMA_PDATAALIGN_BYTE;
510  ctx->hdma_tx.Init.MemDataAlignment = DMA_MDATAALIGN_BYTE;
511  ctx->hdma_tx.Init.Mode = DMA_NORMAL;
512  ctx->hdma_tx.Init.Priority = ctx->dma_tx_priority;
513  ctx->hdma_tx.Init.FIFOMode = DMA_FIFOMODE_DISABLE;
514  ctx->hdma_tx.Init.FIFOThreshold = DMA_FIFO_THRESHOLD_FULL;
515  ctx->hdma_tx.Init.MemBurst = DMA_MBURST_SINGLE;
516  ctx->hdma_tx.Init.PeriphBurst = DMA_MBURST_SINGLE;
517 
518  HAL_DMA_Init(&ctx->hdma_tx);
519 
520  /* Associate the initialized DMA handle to the the SPI handle */
521  __HAL_LINKDMA(hspi, hdmatx, ctx->hdma_tx);

777  * @details Transmits and receives data on SPI bus, with CS already asserted.

779  * @param[in] *inst SPI instance.

781  * @param[in] clk_speed SPI clock speed @ref i2_spi_clock_speed_t.
782  * @param[in] spi_mode SPI communication mode @ref i2_spi_mode_t.
783  * @param[in] first_bit Communication start bit @ref i2_spi_first_bit_t.
784  * @param[in] *txbuf If null, then this function will only receive data on
785  * SPI bus, and put it in rxbuf, else this buffer will be
786  * transmitted on SPI bus.
787  * @param[out] *rxbuf If null, then this function will only transmit data
788  * form txbuf on SPI bus, else this buffer will be used
789  * to receive data on SPI bus.
790  * @param[in] size Amount of data to be transmitted / received.
791  * @param[in] timeout SPI bus HAL timeout.
792  * @return Execution error code @ref I2_ERROR.
793  *
794  * @note If both txbuf and rxbuf are non null, then txbuf will be transmitted
795  * TX pin and on same clock data will be received on RX pin and will be
796  * put in rxbuf.
797  */
798 i2_error i2_spi_txrx_raw( i2_spi_inst_t *inst, i2_spi_data_width_t data_width,
799  i2_spi_clock_speed_t clk_speed,
800  i2_spi_mode_t spi_mode, i2_spi_first_bit_t first_bit,
801  uint8_t *txbuf, uint8_t *rxbuf,
802  int32_t size, uint32_t timeout)
803 {
804  i2_error err = I2_FAILURE;
805  HAL_StatusTypeDef retval;
806  i2_spi_ctx_t *ctx;
807 
808  if ( !inst ) {
809  return I2_INVALID_PARAM;
810  }
811 
812  ctx = spi_inst_to_ctx(inst->ctx_name);
813  if ( !ctx ) {
814  return I2_INVALID_PARAM;
815  }
816 
817  err = spi_config(ctx, data_width, clk_speed, spi_mode, first_bit);
818  if ( err != I2_SUCCESS ) {
819  return err;
820  }
821 
822  if ( ctx->hal_mode != POLLING_MODE ) {
823  ctx->status = I2_TRANSFER_WAIT;
824  }
825 
826  if (ctx->hal_mode == INTERRUPT_MODE) {
827  if ( !rxbuf ) {
828  retval = HAL_SPI_Transmit_IT(&ctx->spi, txbuf, size);
829  } else if ( !txbuf ) {
830  retval = HAL_SPI_Receive_IT(&ctx->spi, rxbuf, size);
831  } else {
832  retval = HAL_SPI_TransmitReceive_IT(&ctx->spi, txbuf, rxbuf, size);
833  }
834  } else if (ctx->hal_mode == DMA_MODE) {
835  if ( !rxbuf ) {
836  retval = HAL_SPI_Transmit_DMA(&ctx->spi, txbuf, size);
837  } else if ( !txbuf ) {
838  retval = HAL_SPI_Receive_DMA(&ctx->spi, rxbuf, size);
839  } else {
840  retval = HAL_SPI_TransmitReceive_DMA(&ctx->spi, txbuf, rxbuf, size);
841  }
842  return I2_NOT_IMPLEMENTED;
843  } else {
844  retval = HAL_SPI_TransmitReceive(&ctx->spi, txbuf, rxbuf, size,
845  (TickType_t) timeout);
846  }
847 
848  if ( ctx->hal_mode != POLLING_MODE ) {
849 #if defined ( ENABLE_RTOS_AWARE_HAL )
850  if (xSemaphoreTake(ctx->sem, (TickType_t)timeout) == pdFALSE) {
851  retval = HAL_TIMEOUT;
852  }
853 #else
854  while (ctx->status == I2_TRANSFER_WAIT);
855 #endif /* ENABLE_RTOS_AWARE_HAL */
856  }
857 
858  err = i2_get_hal_error(retval);
859 
860  /* Check for error interrupt */
861  if ( ctx->status == I2_TRANSFER_ERROR ) {
862  err = I2_FAILURE;
863  } else if ( ctx->status == I2_TRANSFER_WAIT ) {
864  /* Complete interrupt was not received */
865  err = I2_TIMEOUT;
866  } else if ( txbuf && ctx->spi.TxXferCount ) {
867  /* Check if the tx xfer is done */
868  err = I2_FAILURE;
869  } else if ( rxbuf && ctx->spi.RxXferCount ) {
870  /* Check if the rx xfer is done */
871  err = I2_FAILURE;
872  }
873 
874  return err;
875 }
