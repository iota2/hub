536 /**
537  * @brief UART initialization.
538  * @details Initializes UART instance along with GPIO, DMA, FIFO and interrupts.
539  *
540  * @param[in] *inst UART instance to initialize.
541  * @return Initialization error code @ref I2_ERROR.
542  */
543 i2_error i2_uart_init(i2_uart_inst_t *inst)
544 {
545  i2_error retval = I2_SUCCESS;
546  i2_uart_ctx_t *ctx;
547  UART_HandleTypeDef *huart;
548  USART_TypeDef *base;
549  IRQn_Type irqn;
550 
551  uint32_t alt;
552  uint32_t hwcontrol = UART_HWCONTROL_NONE;
553  uint32_t mode = 0;
554 
555  if ( !inst ) {
556  return I2_INVALID_PARAM;
557  }
558 
559  /* get UART context for required name */
560  ctx = uart_get_ctx_from_name(inst->ctx_name);
561  if ( !ctx ) {
562  return I2_INVALID_PARAM;
563  }
564 
565  /* Prevent re-init */
566  if ( ctx->initialized ) {
567  return I2_NOT_AVAILABLE;
568  }
569 
570  /* Get the pointer to the UART handle */
571  huart = &(ctx->uart);
572 
573  /* Get the UART controller base address */
574  base = ctx->baseaddr;
575 
576 #if defined ( ENABLE_RTOS_AWARE_HAL )
577  /* Create mutexes. */
578  ctx->mutex_tx = xSemaphoreCreateMutex();
579  ctx->mutex_rx = xSemaphoreCreateMutex();
580 
581  /* Create semaphores */
582  ctx->sem_tx = xSemaphoreCreateBinary();
583  ctx->sem_rx = xSemaphoreCreateBinary();
584 #endif /* ENABLE_RTOS_AWARE_HAL */
585  ctx->tx_status = I2_TRANSFER_DONE;
586  ctx->rx_status = I2_TRANSFER_DONE;
587 
588  if ( base == USART1 ) {
589 #if defined ( I2_ENABLE_UART1_CONTEXT )
590  usart1 = huart;
591  irqn = USART1_IRQn;
592  alt = GPIO_AF7_USART1; /* USART1 will always associate to AF7 */
593  __HAL_RCC_USART1_CLK_ENABLE();
594  if ( (ctx->rx_hal_mode == DMA_MODE) ||
595  (ctx->tx_hal_mode == DMA_MODE) ) {
596  __HAL_RCC_DMA2_CLK_ENABLE(); /* USART1 will always associate to DMA2 */
597  }
598 #else
599  return I2_NOT_AVAILABLE;
600 #endif /* I2_ENABLE_UART1_CONTEXT */
601  } else if ( base == USART2 ) {
602 #if defined ( I2_ENABLE_UART2_CONTEXT )
603  irqn = USART2_IRQn;
604  usart2 = huart;
605  alt = GPIO_AF7_USART2; /* USART2 will always associate to AF7 */
606  __HAL_RCC_USART2_CLK_ENABLE();
607  if ( (ctx->rx_hal_mode == DMA_MODE) ||
608  (ctx->tx_hal_mode == DMA_MODE) ) {
609  __HAL_RCC_DMA1_CLK_ENABLE();
610  }
611 #else
612  return I2_NOT_AVAILABLE;
613 #endif /* I2_ENABLE_UART2_CONTEXT */
614  } else if ( base == USART3 ) {
615 #if defined ( I2_ENABLE_UART3_CONTEXT )
616  usart3 = huart;
617  irqn = USART3_IRQn;
618  alt = GPIO_AF7_USART3; /* USART3 will always associate to AF7 */
619  __HAL_RCC_USART3_CLK_ENABLE();
620  if ( (ctx->rx_hal_mode == DMA_MODE) ||
621  (ctx->tx_hal_mode == DMA_MODE) ) {
622  __HAL_RCC_DMA1_CLK_ENABLE();
623  }
624 #else
625  return I2_NOT_AVAILABLE;
626 #endif /* I2_ENABLE_UART3_CONTEXT */
627  } else if ( base == UART4 ) {
628 #if defined ( I2_ENABLE_UART4_CONTEXT )
629  uart4 = huart;
630  irqn = UART4_IRQn;
631  alt = GPIO_AF8_UART4; /* UART4 will always associate to AF8 */
632  __HAL_RCC_UART4_CLK_ENABLE();
633  if ( (ctx->rx_hal_mode == DMA_MODE) ||
634  (ctx->tx_hal_mode == DMA_MODE) ) {
635  __HAL_RCC_DMA1_CLK_ENABLE();
636  }
637 #else
638  return I2_NOT_AVAILABLE;
639 #endif /* I2_ENABLE_UART4_CONTEXT */
640  } else if ( base == UART5 ) {
641 #if defined ( I2_ENABLE_UART5_CONTEXT )
642  uart5 = huart;
643  irqn = UART5_IRQn;
644  alt = GPIO_AF8_UART5; /* UART5 will always associate to AF8 */
645  __HAL_RCC_UART5_CLK_ENABLE();
646  if ( (ctx->rx_hal_mode == DMA_MODE) ||
647  (ctx->tx_hal_mode == DMA_MODE) ) {
648  __HAL_RCC_DMA1_CLK_ENABLE();
649  }
650 #else
651  return I2_NOT_AVAILABLE;
652 #endif /* I2_ENABLE_UART5_CONTEXT */
653  } else if ( base == USART6 ) {
654 #if defined ( I2_ENABLE_UART6_CONTEXT )
655  usart6 = huart;
656  irqn = USART6_IRQn;
657  alt = GPIO_AF8_USART6; /* USART6 will always associate to AF8 */
658  __HAL_RCC_USART6_CLK_ENABLE();
659  if ( (ctx->rx_hal_mode == DMA_MODE) ||
660  (ctx->tx_hal_mode == DMA_MODE) ) {
661  __HAL_RCC_DMA2_CLK_ENABLE();
662  }
663 #else
664  return I2_NOT_AVAILABLE;
665 #endif /* I2_ENABLE_UART6_CONTEXT */
666  } else {
667  return I2_INVALID_PARAM;
668  }
669 
670  /* UART driver needs gpio */
671  i2_gpio_init();
672 
673  /* Configure TX */
674  if ( i2_gpio_is_valid(&ctx->tx) ) {
675  i2_gpio_config_alt(&ctx->tx, GPIO_MODE_AF_PP, GPIO_PULLUP, alt);
676  mode |= UART_MODE_TX;
677  }
678 
679  /* Configure RX */
680  if ( i2_gpio_is_valid(&ctx->rx) ) {
681  i2_gpio_config_alt(&ctx->rx, GPIO_MODE_AF_PP, GPIO_PULLUP, alt);
682  mode |= UART_MODE_RX;
683  }
684 
685  /* Configure CTS */
686  if ( i2_gpio_is_valid(&ctx->cts) ) {
687  i2_gpio_config_alt(&ctx->cts, GPIO_MODE_AF_OD, GPIO_PULLUP, alt);
688  hwcontrol |= UART_HWCONTROL_CTS;
689  }
690 
691  /* Configure RTS */
692  if ( i2_gpio_is_valid(&ctx->rts) ) {
693  i2_gpio_config_alt(&ctx->rts, GPIO_MODE_AF_OD, GPIO_PULLUP, alt);
694  hwcontrol |= UART_HWCONTROL_RTS;
695  }
696 
697  /* DMA RX Config */
698  if (ctx->rx_hal_mode == DMA_MODE) {
699  /* Configure the DMA handler for reception process */
700  ctx->hdma_rx.Instance = ctx->dma_rx_stream;
701  ctx->hdma_rx.Init.Channel = ctx->dma_rx_channel;
702  ctx->hdma_rx.Init.Direction = DMA_PERIPH_TO_MEMORY;
703  ctx->hdma_rx.Init.PeriphInc = DMA_PINC_DISABLE;
704  ctx->hdma_rx.Init.MemInc = DMA_MINC_ENABLE;
705  ctx->hdma_rx.Init.PeriphDataAlignment = DMA_PDATAALIGN_BYTE;
706  ctx->hdma_rx.Init.MemDataAlignment = DMA_MDATAALIGN_BYTE;
707  ctx->hdma_rx.Init.Mode = DMA_CIRCULAR;
708  ctx->hdma_rx.Init.Priority = ctx->dma_rx_priority;
709  ctx->hdma_rx.Init.FIFOMode = DMA_FIFOMODE_DISABLE;
710  ctx->hdma_rx.Init.FIFOThreshold = DMA_FIFO_THRESHOLD_FULL;
711  ctx->hdma_rx.Init.MemBurst = DMA_MBURST_SINGLE;
712  ctx->hdma_rx.Init.PeriphBurst = DMA_PBURST_SINGLE;
713 
714  HAL_DMA_Init(&ctx->hdma_rx);
715 
716  /* Associate the initialized DMA handle to the the UART handle */
717  __HAL_LINKDMA(huart, hdmarx, ctx->hdma_rx);
718 
719  /* NVIC configuration for DMA transfer complete interrupt */
720  HAL_NVIC_SetPriority(ctx->dma_rx_irqn, UART_DMA_PREEMPTION_PRIORITY,
721  UART_DMA_SUB_PRIORITY);
722  HAL_NVIC_EnableIRQ(ctx->dma_rx_irqn);
723  }
724 
725  /* DMA TX Config */
726  if (ctx->tx_hal_mode == DMA_MODE) {
727  /* Configure the DMA handler for Transmission process */
728  ctx->hdma_tx.Instance = ctx->dma_tx_stream;
729  ctx->hdma_tx.Init.Channel = ctx->dma_tx_channel;
730  ctx->hdma_tx.Init.Direction = DMA_MEMORY_TO_PERIPH;
731  ctx->hdma_tx.Init.PeriphInc = DMA_PINC_DISABLE;
732  ctx->hdma_tx.Init.MemInc = DMA_MINC_ENABLE;
733  ctx->hdma_tx.Init.PeriphDataAlignment = DMA_PDATAALIGN_BYTE;
734  ctx->hdma_tx.Init.MemDataAlignment = DMA_MDATAALIGN_BYTE;
735  ctx->hdma_tx.Init.Mode = DMA_NORMAL;
736  ctx->hdma_tx.Init.Priority = ctx->dma_tx_priority;
737  ctx->hdma_tx.Init.FIFOMode = DMA_FIFOMODE_DISABLE;
738  ctx->hdma_tx.Init.FIFOThreshold = DMA_FIFO_THRESHOLD_FULL;
739  ctx->hdma_tx.Init.MemBurst = DMA_MBURST_SINGLE;
740  ctx->hdma_tx.Init.PeriphBurst = DMA_PBURST_SINGLE;
741 
742  HAL_DMA_Init(&ctx->hdma_tx);
743 
744  /* Associate the initialized DMA handle to the UART handle */
745  __HAL_LINKDMA(huart, hdmatx, ctx->hdma_tx);
746 
747  /* NVIC configuration for DMA transfer complete interrupt */
748  HAL_NVIC_SetPriority(ctx->dma_tx_irqn, UART_DMA_PREEMPTION_PRIORITY,
749  UART_DMA_SUB_PRIORITY);
750  HAL_NVIC_EnableIRQ(ctx->dma_tx_irqn);
751  }
752 
753  if ((ctx->rx_hal_mode == DMA_MODE) || (ctx->rx_hal_mode == INTERRUPT_MODE) ||
754  (ctx->tx_hal_mode == DMA_MODE) || (ctx->tx_hal_mode == INTERRUPT_MODE) ) {
755  /* NVIC configuration for UART */
756  HAL_NVIC_SetPriority(irqn, UART_PREEMPTION_PRIORITY, UART_SUB_PRIORITY);
757  HAL_NVIC_EnableIRQ(irqn);
758  }
759 
760  /* UART configuration */
761  huart->Instance = base;
762  huart->Init.BaudRate = ctx->baud_rate;
763  huart->Init.HwFlowCtl = hwcontrol;
764  huart->Init.Mode = mode;
765  /* Assume these values can be fixed for all UART for now */
766  huart->Init.WordLength = UART_WORDLENGTH_8B;
767  huart->Init.StopBits = UART_STOPBITS_1;
768  huart->Init.Parity = UART_PARITY_NONE;
769  huart->Init.OverSampling = UART_OVERSAMPLING_8;
770 
771  if (HAL_UART_Init(huart) != HAL_OK) {
772  retval = I2_FAILURE;
773  } else {
774  int32_t i;
775  for ( i = 0; i < I2_MAX_NUM_UART_CONTEXT; i++ ) {
776  if ( !i2_uart_inst_table[i] ) {
777  i2_uart_inst_table[i] = inst;
778  break;
779  }
780  }
781  }
782 
783  i2_fifo_init(&ctx->rx_fifo, ctx->buff, I2_UART_FIFO_SIZE);
784 
785  ctx->initialized = true;
786  ctx->rx_trigger_level = I2_UART_FIFO_HALF_SIZE;
787  ctx->rx_buffering_on = false;
788  ctx->rx_water_mark = 0;
789 
790  return retval;
791 }

860 /**
861  * @brief UART transmission function.
862  * @details The interrupt/DMA based UARt transmission functions.
863  *
864  * @param[in] *inst UART instance to use.
865  * @param[in] *txbuf buffer to transmit.
866  * @param[in] size Number of bytes to send.
867  * @param[out] *num_bytes Number of bytes actually sent.
868  * @param[in] *timeout UART peripheral HAL timeput.
869  * @return Execution error code @ref I2_ERROR.
870  *
871  * @note It will block until the UART TX is finished.
872  */
873 i2_error i2_uart_tx(i2_uart_inst_t *inst, uint8_t *txbuf, int32_t size,
874  int32_t *num_bytes, uint32_t timeout)
875 {
876  i2_error err = I2_FAILURE;
877  HAL_StatusTypeDef retval;
878  i2_uart_ctx_t *ctx;
879  UART_HandleTypeDef *huart;
880 
881  if ( !inst ) {
882  return I2_INVALID_PARAM;
883  }
884 
885  ctx = uart_get_ctx_from_name(inst->ctx_name);
886  if ( !ctx ) {
887  return I2_INVALID_PARAM;
888  }
889 
890 #if defined( ENABLE_RTOS_AWARE_HAL )
891  if (xSemaphoreTake(ctx->mutex_tx, (TickType_t)timeout) == pdFALSE) {
892  if ( num_bytes ) {
893  *num_bytes = 0;
894  }
895  return I2_TIMEOUT;
896  }
897 #endif /* ENABLE_RTOS_AWARE_HAL */
898 
899  /* Get the pointer to the UART handle */
900  huart = &(ctx->uart);
901 
902  ctx->tx_status = I2_TRANSFER_WAIT;
903 
904  if (ctx->tx_hal_mode == INTERRUPT_MODE) {
905  retval = HAL_UART_Transmit_IT(huart, txbuf, size);
906  } else if (ctx->tx_hal_mode == DMA_MODE) {
907  retval = HAL_UART_Transmit_DMA(huart, txbuf, size);
908  } else {
909  err = I2_NOT_SUPPORTED;
910  goto err;
911  }
912 
913 #if defined ( ENABLE_RTOS_AWARE_HAL )
914  if (xSemaphoreTake(ctx->sem_tx, (TickType_t)timeout) == pdFALSE) {
915  retval = HAL_TIMEOUT;
916  } /* ENABLE_RTOS_AWARE_HAL */
917 #else
918  while (ctx->tx_status == I2_TRANSFER_WAIT);
919 #endif
920 
921  err = i2_get_hal_error(retval);
922 
923  /* Check for error interrupt */
924  if ( ctx->tx_status == I2_TRANSFER_ERROR ) {
925  err = I2_FAILURE;
926  } else if (ctx->tx_status == I2_TRANSFER_WAIT) {
927  /* Tx Complete interrupt was not received */
928  err = I2_TIMEOUT;
929  }
930 
931  /* Record correct number of sent bytes */
932  if ( num_bytes ) {
933  if ((err == I2_SUCCESS) || (err == I2_TIMEOUT)) {
934  *num_bytes = (size - huart->TxXferCount);
935  } else {
936  *num_bytes = 0;
937  }
938  }
939 
940  if ( err != I2_SUCCESS ) {
941  HAL_UART_AbortTransmit(huart);
942  }
943 
944 err:
945 #if defined ( ENABLE_RTOS_AWARE_HAL )
946  xSemaphoreGive(ctx->mutex_tx);
947 #endif /* ENABLE_RTOS_AWARE_HAL */
948 
949  return err;
950 }

1112 /**
1113  * @brief UART receiving function.
1114  * @details The interrupt/DMA based UART reception can be called by users after
1115  * UART RX buffering starts.
1116  * Let's say this function is called with the caller buffer of size X bytes
1117  * and time out T msec.
1118  * If there is already *any* number of bytes in the UART driver's receive FIFO,
1119  * this function returns right away regardless of X or T values.
1120  * If the UART driver's receive FIFO is empty at the time of calling this
1121  * function the caller will block on UART RX semaphore until one of the
1122  * following conditions is met:
1123  * a) UART driver's receive FIFO gets *any* number of bytes, followed by UART RX
1124  * line going idle regardless of X or T values.
1125  * b) UART driver's receive FIFO gets filled up to half full without the UART RX
1126  * line going idle regardless of X or T values.
1127  * c) UART driver does not get any byte in the UART driver's receive FIFO until
1128  * the timeout expires, then this funciton returns with no data.
1129  * It's indicated by the *num_bytes being zero and also by the return value.
1130  * In a) and b) cases, if the caller's buffer size X happens to be smaller than
1131  * the available number of UART RX bytes in the UART driver's receive FIFO, then
1132  * the UART driver copies X number of bytes first and the caller must call
1133  * this function again to read in the remaining bytes.
1134  * It's important to remember whenever UART RX line goes idle after receiving
1135  * any number of bytes, this function will return. This is done to meet UART RX
1136  * responsiveness requirement, as UART driver can't wait until either buffer
1137  * gets full or timeout happens from the performance point of view.
1138  * Also note that if the caller does not read out data in UART receive FIFO
1139  * fast enough, the buffered data in the FIFO will start to be overwritten
1140  * by new data.
1141  *
1142  * @param[in] *inst UART instance to use.
1143  * @param[in] *rxbuf buffer to receive data in.
1144  * @param[in] size Number of bytes to receive.
1145  * @param[out] *num_bytes Number of bytes actually received.
1146  * @param[in] *timeout UART peripheral HAL timeput.
1147  * @return Execution error code @ref I2_ERROR.
1148  */
1149 i2_error i2_uart_rx(i2_uart_inst_t *inst, uint8_t *rxbuf, int32_t size,
1150  int32_t *num_bytes, uint32_t timeout)
1151 {
1152  i2_error err = I2_FAILURE;
1153  i2_uart_ctx_t *ctx;
1154  int32_t _size;
1155 
1156  if ( !inst ) {
1157  return I2_INVALID_PARAM;
1158  }
1159 
1160  ctx = uart_get_ctx_from_name(inst->ctx_name);
1161  if ( !ctx ) {
1162  return I2_INVALID_PARAM;
1163  }
1164 
1165 #if defined ( ENABLE_RTOS_AWARE_HAL )
1166  /* Make sure one thread uses one controller at a time */
1167  if (xSemaphoreTake(ctx->mutex_rx, (TickType_t)timeout) == pdFALSE) {
1168  if ( num_bytes ) {
1169  *num_bytes = 0;
1170  }
1171  return I2_TIMEOUT;
1172  }
1173 #endif /* ENABLE_RTOS_AWARE_HAL */
1174 
1175  /* desired read size */
1176  _size = size;
1177 
1178  /* If the FIFO is empty, wait for signal */
1179  if ( !i2_fifo_count(&ctx->rx_fifo, false) ) {
1180  ctx->rx_status = I2_TRANSFER_WAIT;
1181  #if defined ( ENABLE_RTOS_AWARE_HAL)
1182  if (xSemaphoreTake(ctx->sem_rx, (TickType_t)timeout) == pdFALSE) {
1183  err = I2_TIMEOUT;
1184  }
1185 #else /* ENABLE_RTOS_AWARE_HAL */
1186  while (ctx->rx_status == I2_TRANSFER_WAIT);
1187 #endif
1188  }
1189 
1190  /* RX ISR says the FIFO is at least half full or rx line is idle */
1191  if ( err != I2_TIMEOUT ) {
1192  i2_fifo_copy(&ctx->rx_fifo, rxbuf, &_size, false);
1193  } else {
1194  _size = 0;
1195  }
1196 
1197  /* Record correct number of received bytes */
1198  if ( num_bytes ) {
1199  *num_bytes = _size;
1200  }
1201 
1202  /* If we return without data, it's a timeout */
1203  if ( _size ) {
1204  err = I2_SUCCESS;
1205  } else {
1206  err = I2_TIMEOUT;
1207  }
1208 
1209 #if defined ( ENABLE_RTOS_AWARE_HAL )
1210  xSemaphoreGive(ctx->mutex_rx);
1211 #endif /* ENABLE_RTOS_AWARE_HAL */
1212 
1213  return err;
1214 }
